## Orange UI for react

### 启动

```html
<!-- 开发环境 -->
yarn dev
```

```html
<!-- npm -->
yarn npm
yarn login
yarn publish
```


### 如何使用

```html
<!-- 推荐使用yarn -->
yarn add orange-ui-react

import { Button } from 'orange-ui-react'
import { Navbar } from 'orange-ui-react'
<!-- 目前只有4个小图标 -->
import { Icon } from 'orange-ui-react'
<!-- 仿头条下拉刷新以及加载更多 -->
import { Refresh } from 'orange-ui-react'

```
